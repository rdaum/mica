verb sync_view_revision(view)
  if ui/View(view) == false
    return 0
  end

  return ui/status_revision()
end

verb sync_view_tree(view, revision)
  let current = ui/current_actor()
  if ui/View(view) == false || current == nothing
    return ui/login_view_node()
  end

  return ui/world_view_node(current, revision, 40)
end
